```python
"""NoneBot2 入口 — MemePilot QQ 表情包机器人。

启动流程：
1. 初始化 NoneBot2 框架（fastapi 驱动器 + OneBot V11 适配器）
2. 注册 startup hook：初始化 engine 服务并执行首次索引同步
3. 加载 bot/plugins/ 下所有命令插件
4. 启动驱动器监听反向 WebSocket
"""

import logging
import os
from pathlib import Path

import nonebot
from nonebot.adapters.onebot.v11 import Adapter as OneBotV11Adapter

from bot.app_state import init_app
from bot.config import MEMES_DIR
from bot.engine import (
    AIMatcher,
    DeepSeekOcrService,
    EmbeddingService,
    ImageOptimizer,
    IndexManager,
    KeywordSearcher,
    RerankService,
)
from bot.logging_config import setup_logging

logger = logging.getLogger(__name__)

# 项目根目录（与 config.py 中的定位方式一致）
_PROJECT_ROOT = MEMES_DIR.parent


def _read_sync_concurrency() -> int | None:
    """从环境变量读取索引同步并发上限。"""
    raw = os.environ.get("SYNC_CONCURRENCY", "")
    if not raw:
        return None
    try:
        value = int(raw)
        return value if value > 0 else None
    except ValueError:
        return None


async def _on_startup() -> None:
    """NoneBot2 启动钩子 — 初始化引擎服务并执行首次索引同步。"""
    setup_logging("log")
    logger.info("MemePilot 正在启动...")

    ocr_service = DeepSeekOcrService()
    embedding_service = EmbeddingService()
    rerank_service = RerankService()
    image_optimizer = ImageOptimizer()

    data_dir = str(_PROJECT_ROOT / "data")
    memes_dir = str(MEMES_DIR)
    sync_concurrency = _read_sync_concurrency()

    index_manager = IndexManager(
        data_dir=data_dir,
        memes_dir=memes_dir,
        ocr_provider=ocr_service,
        embedding_provider=embedding_service,
        sync_concurrency=sync_concurrency,
        optimizer=image_optimizer,
    )
    index_manager.load()

    logger.info("开始首次索引同步...")
    try:
        result = await index_manager.sync_with_filesystem()
    except Exception:
        logger.exception("首次索引同步失败，Bot 无法启动")
        raise RuntimeError("首次索引同步失败") from None

    logger.info(
        "索引同步完成: 新增=%d, 删除=%d, 去重=%d, 无文字移走=%d, 失败=%d",
        result.added, result.deleted, result.deduped,
        result.no_text_moved, len(result.failed),
    )

    if index_manager.entry_count == 0 and result.added == 0:
        logger.warning("表情包目录为空，Bot 启动但搜索功能不可用")

    ai_matcher = AIMatcher(
        index_provider=index_manager,
        embedding_provider=embedding_service,
        rerank_provider=rerank_service,
    )
    keyword_searcher = KeywordSearcher(index_provider=index_manager)

    init_app(
        index_manager=index_manager,
        ocr_service=ocr_service,
        embedding_service=embedding_service,
        image_optimizer=image_optimizer,
        ai_matcher=ai_matcher,
        keyword_searcher=keyword_searcher,
    )

    logger.info("MemePilot 启动完成")


def main() -> None:
    """NoneBot2 主入口。"""
    nonebot.init()

    driver = nonebot.get_driver()
    driver.on_startup(_on_startup)
    driver.register_adapter(OneBotV11Adapter)

    nonebot.load_plugins("bot/plugins")

    host = os.environ.get("BOT_HOST", "0.0.0.0")
    port = int(os.environ.get("BOT_PORT", "8080"))
    nonebot.run(host=host, port=port)


if __name__ == "__main__":
    main()
```

"""bot.bot 模块单元测试。"""

import os
from unittest.mock import patch

from bot.bot import _read_sync_concurrency


class TestReadSyncConcurrency:
    """_read_sync_concurrency() 测试。"""

    def test_returns_none_when_unset(self):
        """未设置 SYNC_CONCURRENCY 时返回 None。"""
        with patch.dict(os.environ, {}, clear=False):
            os.environ.pop("SYNC_CONCURRENCY", None)
            assert _read_sync_concurrency() is None

    def test_returns_none_when_empty(self):
        """SYNC_CONCURRENCY 为空字符串时返回 None。"""
        with patch.dict(os.environ, {"SYNC_CONCURRENCY": ""}):
            assert _read_sync_concurrency() is None

    def test_returns_value_when_valid(self):
        """SYNC_CONCURRENCY 为有效正整数时返回该值。"""
        with patch.dict(os.environ, {"SYNC_CONCURRENCY": "10"}):
            assert _read_sync_concurrency() == 10

    def test_returns_none_when_zero(self):
        """SYNC_CONCURRENCY 为 0 时返回 None。"""
        with patch.dict(os.environ, {"SYNC_CONCURRENCY": "0"}):
            assert _read_sync_concurrency() is None

    def test_returns_none_when_negative(self):
        """SYNC_CONCURRENCY 为负数时返回 None。"""
        with patch.dict(os.environ, {"SYNC_CONCURRENCY": "-3"}):
            assert _read_sync_concurrency() is None

    def test_returns_none_when_not_integer(self):
        """SYNC_CONCURRENCY 非整数时返回 None。"""
        with patch.dict(os.environ, {"SYNC_CONCURRENCY": "abc"}):
            assert _read_sync_concurrency() is None
